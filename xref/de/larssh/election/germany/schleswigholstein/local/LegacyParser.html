103 		try (BufferedReader bufferedReader = new BufferedReader(reader)) {
104 			final AtomicReference<OptionalInt> numberOfAllBallots = new AtomicReference<>(OptionalInt.empty());
105 			final List<LocalBallot> ballots = new ArrayList<>();
106 			bufferedReader.lines().map(String::trim).filter(line -> !line.isEmpty() && line.charAt(0) != LINE_COMMENT).forEachOrdered(line -> {
107 				if (line.charAt(0) == LINE_COMMAND) {
108 					final String command = Strings.trimStart(line.substring(1));
109 					if (COMMAND_CLEAR.equalsIgnoreCase(command)) {
110 						ballots.clear();
111 					}
112 					Patterns.matches(NUMBER_OF_ALL_BALLOTS_PATTERN, command).ifPresent(matcher -> numberOfAllBallots.set(OptionalInt.of(Integer.parseInt(matcher.group(GROUP_VALUE)))));
113 				} else {
114 					ballots.addAll(createBallotFromLine(election, pollingStation, line));
115 				}
116 			});
117 			return new LocalElectionResult(election, numberOfAllBallots.get(), ballots, emptySet(), emptySet());
