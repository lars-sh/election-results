8   public class PartyController extends PartyUiController {
9   	private static de.larssh.election.germany.schleswigholstein.Color colorFromJavaFx(final Color color) {
10  		return new de.larssh.election.germany.schleswigholstein.Color(color.getRed(),
11  				color.getGreen(),
12  				color.getBlue(),
13  				color.getOpacity());
14  	}
15  
16  	private static Color colorToJavaFx(final de.larssh.election.germany.schleswigholstein.Color color) {
17  		return new Color(color.getRed(), color.getGreen(), color.getBlue(), color.getOpacity());
18  	}
19  
20  	public PartyController(final LocalElectionController parent) {
21  		super(parent);
22  	}
23  
24  	public Party getParty() {
25  		return new Party(getShortName().getText(),
26  				getName().getText(),
27  				getType().getValue(),
28  				colorFromJavaFx(getBackgroundColor().getValue()),
29  				colorFromJavaFx(getFontColor().getValue()));
30  	}
31  
32  	@Override
33  	protected void initialize() {
34  		getType().setItems(FXCollections.observableArrayList(PartyType.values()));
35  
36  		reset();
37  	}
38  
39  	public void reset() {
40  		getShortName().setText("");
41  		getName().setText("");
42  		getType().setValue(PartyType.POLITICAL_PARTY);
43  		getBackgroundColor().setValue(Color.WHITE);
44  		getFontColor().setValue(Color.BLACK);
45  	}
46  
47  	public void setParty(final Party party) {
48  		getShortName().setText(party.getShortName());
49  		getName().setText(party.getName());
50  		getType().setValue(party.getType());
51  		getBackgroundColor().setValue(colorToJavaFx(party.getBackgroundColor()));
52  		getFontColor().setValue(colorToJavaFx(party.getFontColor()));
53  	}
54  }
