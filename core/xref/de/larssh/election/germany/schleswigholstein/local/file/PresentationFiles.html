162 		 * Returns the exact number of all ballots if available. Else the number of all
163 		 * ballots is estimated.
164 		 *
165 		 * @param pollingStations the polling stations
166 		 * @return the estimated number of all ballots
167 		 */
168 		@SuppressWarnings("checkstyle:MagicNumber")
169 		private BigDecimal estimateNumberOfAllBallots(final Set<LocalPollingStation> pollingStations) {
170 			// Start by calculating the known numbers of ballots
171 			int numberOfPresent = 0;
172 			int sumOfAllBallots = 0;
173 			for (final LocalPollingStation pollingStation : pollingStations) {
174 				final OptionalInt numberOfAllBallots = result.getNumberOfAllBallots(pollingStation);
175 				if (numberOfAllBallots.isPresent()) {
176 					sumOfAllBallots += numberOfAllBallots.getAsInt();
177 					numberOfPresent += 1;
178 				}
179 			}
180 			// In case no information about the number of ballots is available, let's
181 			// estimate exactly one ballot per polling station.
182 			if (numberOfPresent == 0) {
183 				return BigDecimal.valueOf(pollingStations.size() - numberOfPresent);
184 			}
185 			// Increase the sum of known ballots proportionally
186 			return BigDecimals.divide((long) sumOfAllBallots * pollingStations.size(), numberOfPresent, 3);
187 		}
