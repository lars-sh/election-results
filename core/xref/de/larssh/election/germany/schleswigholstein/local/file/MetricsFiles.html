301 		/**
302 		 * Returns a {@link CellStyle} instance with the coloring of {@code party} and
303 		 * {@code dataFormat}. Cell styles are cached and reused.
304 		 *
305 		 * @param workbook   the current workbook
306 		 * @param party      the {@link Party} to be used for coloring or empty
307 		 * @param dataFormat the Excel data format to set or empty
308 		 * @return the {@link CellStyle} instance with the coloring of {@code party} and
309 		 *         {@code dataFormat}
310 		 */
311 		private Optional<CellStyle> getCellStyle(final Workbook workbook, final Optional<Party> party, final Optional<String> dataFormat) {
312 			if (!party.isPresent() && !dataFormat.isPresent()) {
313 				return Optional.empty();
314 			}
315 			final String key = Keys.escape(party.map(Party::getKey).orElse(""), ", ", dataFormat.orElse(""), "");
316 			return Optional.of(cellStyleCache.computeIfAbsent(key, k -> {
317 				final CellStyle cellStyle = workbook.createCellStyle();
318 				party.ifPresent(p -> {
319 					cellStyle.setFillPattern(FillPatternType.SOLID_FOREGROUND);
320 					final Color backgroundColor = p.getBackgroundColor();
321 					cellStyle.setFillForegroundColor(new XSSFColor(new byte[] {(byte) backgroundColor.getAlphaAsByte(), (byte) backgroundColor.getRedAsByte(), (byte) backgroundColor.getGreenAsByte(), (byte) backgroundColor.getBlueAsByte()}));
322 					final Color fontColor = p.getFontColor();
323 					final XSSFFont font = (XSSFFont) workbook.createFont();
324 					font.setColor(new XSSFColor(new byte[] {(byte) fontColor.getAlphaAsByte(), (byte) fontColor.getRedAsByte(), (byte) fontColor.getGreenAsByte(), (byte) fontColor.getBlueAsByte()}));
325 					cellStyle.setFont(font);
326 				});
327 				dataFormat.ifPresent(f -> cellStyle.setDataFormat(workbook.createDataFormat().getFormat(f)));
328 				return cellStyle;
329 			}));
330 		}

332 		/**
333 		 * Formats and writes {@link #result} to {@link #writer}.
334 		 *
335 		 * @throws IOException on IO error
336 		 */
337 		@PackagePrivate
338 		void write() throws IOException {
339 			try (XSSFWorkbook workbook = new XSSFWorkbook()) {
340 				workbook.setCellFormulaValidation(false);
341 				writeOverview(workbook.createSheet("Übersicht"));
342 				writeParties(workbook.createSheet("Gruppierungen"));
343 				writeBlockVotes(workbook.createSheet("Blockstimmen"));
344 				writeNominations(workbook.createSheet("Kandidierende"));
345 				// Auto Size Columns
346 				workbook.sheetIterator().forEachRemaining(sheet -> {
347 					final int numberOfColumns = sheet.getRow(0).getLastCellNum();
348 					for (int columnIndex = 0; columnIndex < numberOfColumns; columnIndex += 1) {
349 						sheet.autoSizeColumn(columnIndex);
350 						// Add the width of the auto filter
351 						final double widthOfHeader = SheetUtil.getColumnWidth(sheet, columnIndex, false, 0, 0);
352 						if (widthOfHeader != -1) {
353 							final int intWidth = (int) Math.round(Math.min(CHARACTER_WIDTH * widthOfHeader + AUTO_FILTER_WIDTH, COLUMN_MAX_WIDTH));
354 							if (intWidth > sheet.getColumnWidth(columnIndex)) {
355 								sheet.setColumnWidth(columnIndex, intWidth);
356 							}
357 						}
358 					}
359 				});
360 				workbook.write(outputStream);
361 			}
362 		}

467 		 *
468 		 * @param row   the row to write to
469 		 * @param party the party
470 		 */
471 		@SuppressFBWarnings(value = "OI_OPTIONAL_ISSUES_PRIMITIVE_VARIANT_PREFERRED", justification = "need to match generics")
472 		private void writeParty(final Row row, final Party party) {
473 			final Optional<Party> optionalParty = Optional.of(party);
474 			// Gruppierung
475 			appendString(row, optionalParty, party.getShortName());
476 			for (final LocalPollingStation pollingStation : result.getElection().getPollingStations()) {
477 				// Stimmen
478 				appendNumber(row, optionalParty, Optional.of(result.filterByDistrict(pollingStation).getPartyResults().get(party).getNumberOfVotes()));
479 				// Stimmen %
480 				appendFormula(row, optionalParty, DATA_FORMAT_PERCENTAGE, "Gruppierungen[[#This Row],[" + pollingStation.getName() + "]] / SUM(Gruppierungen[" + pollingStation.getName() + "])");
481 			}
482 			// Gesamt
483 			appendFormula(row, optionalParty, result.getElection().getPollingStations().stream().map(LocalPollingStation::getName).collect(joining("] + Gruppierungen[", "Gruppierungen[", "]")));
484 			// Gesamt %
485 			appendFormula(row, optionalParty, DATA_FORMAT_PERCENTAGE, "Gruppierungen[[#This Row],[Gesamt]] / SUM(Gruppierungen[Gesamt])");
486 			// Sitze
487 			appendFormula(row, optionalParty, "COUNTIFS(Kandidierende[Gruppierung], Gruppierungen[[#This Row],[Gruppierung]], Kandidierende[Mandat], \"<>\")");
488 		}
