185 		/**
186 		 * Tries to parse {@code link}.
187 		 *
188 		 * @param lineNumber the line number
189 		 * @param line       the line content
190 		 */
191 		@SuppressWarnings({"checkstyle:XIllegalCatchDefault", "PMD.AvoidCatchingGenericException"})
192 		private void parseLine(final int lineNumber, final String line) {
193 			// Ballots
194 			if (line.charAt(0) != LINE_COMMAND) {
195 				try {
196 					ballots.addAll(createBallotsFromLine(line));
197 				} catch (final Exception e) {
198 					exceptions.add(new PollingStationResultFileLineParseException(e, lineNumber, line, e.getMessage()));
199 				}
200 				return;
201 			}
202 			// Clear Command
203 			final String command = line.substring(1).trim();
204 			if (Strings.equalsIgnoreCaseAscii(COMMAND_CLEAR, command)) {
205 				ballots.clear();
206 				return;
207 			}
208 			// Number of all Ballots Command
209 			final Optional<Matcher> matcher = Patterns.matches(NUMBER_OF_ALL_BALLOTS_PATTERN, command);
210 			if (matcher.isPresent()) {
211 				numberOfAllBallots.set(OptionalInt.of(Integer.parseInt(matcher.get().group(GROUP_VALUE))));
212 				return;
213 			}
214 			exceptions.add(new PollingStationResultFileLineParseException(lineNumber, line, "Failed parsing command line."));
215 		}

218 		 * Parses a single {@code line} and creates ballots out of it.
219 		 *
220 		 * @param line the line to create a set of {@link LocalBallot} of
221 		 * @return Stimmzettel
222 		 */
223 		private Collection<LocalBallot> createBallotsFromLine(final String line) {
224 			final Matcher matcher = Patterns.matches(BALLOT_PATTERN, line).orElseThrow(() -> new ElectionException("Failed parsing line \"%s\".", line));
225 			final LocalBallot ballot;
226 			if (BALLOT_INVALID.equals(matcher.group(GROUP_VALUE))) {
227 				ballot = LocalBallot.createInvalidBallot(election, pollingStation, false);
228 			} else {
229 				final Set<LocalNomination> nominations = Arrays.stream(matcher.group(GROUP_VALUE).split("\\s+", SplitLimit.NO_LIMIT)).map(this::findNomination).collect(toSet());
230 				ballot = LocalBallot.createValidBallot(election, pollingStation, false, nominations);
231 			}
232 			final int count = Optional.ofNullable(matcher.group(GROUP_COUNT)).map(Integer::parseInt).orElse(1);
233 			return IntStream.range(0, count).mapToObj(index -> ballot).collect(toList());
234 		}

237 		 * Determines the correct nomination for {@code person}.
238 		 *
239 		 * <p>
240 		 * This method takes care of the party prefix.
241 		 *
242 		 * @param person Bewerberin oder Bewerber
243 		 * @return Bewerberin oder Bewerber
244 		 */
245 		private LocalNomination findNomination(final String person) {
246 			final LocalDistrict district = pollingStation.getDistrict();
247 			if (person.isEmpty()) {
248 				throw new ElectionException("Cannot find a nomination for an empty string in district \"%s\".", district.getName());
249 			}
250 			final char possiblePartyIndicator = Character.toLowerCase(person.charAt(0));
251 			final Set<LocalNomination> nominationsWithPartyPrefix = election.getNominations().stream().filter(nomination -> nomination.getDistrict().equals(district) && nomination.getType() == LocalNominationType.DIRECT && nomination.getParty().isPresent() && !nomination.getParty().get().getShortName().isEmpty() && possiblePartyIndicator == Character.toLowerCase(nomination.getParty().get().getShortName().charAt(0)) && matches(nomination, person.substring(1))).collect(toSet());
252 			if (nominationsWithPartyPrefix.size() == 1) {
253 				return nominationsWithPartyPrefix.iterator().next();
254 			}
255 			final Set<LocalNomination> nominations = election.getNominations().stream().filter(nomination -> nomination.getDistrict().equals(district) && nomination.getType() == LocalNominationType.DIRECT && matches(nomination, person)).collect(toSet());
256 			if (nominations.isEmpty()) {
257 				throw new ElectionException("Cannot find a nomination for \"%s\" in district \"%s\".", person, district.getName());
258 			}
259 			if (nominations.size() > 1) {
260 				throw new ElectionException("Found %d nominations for \"%s\" in district \"%s\": \"%s\"", nominations.size(), person, district.getName(), nominations.stream().map(LocalNomination::getKey).collect(joining("\", \"")));
261 			}
262 			return nominations.iterator().next();
263 		}

266 		 * Checks if {@code nomination} is the correct nomination for {@code person}.
267 		 *
268 		 * <p>
269 		 * This method makes sure, that both, the given and the family name can be in
270 		 * either first or second place of order.
271 		 *
272 		 * @param nomination Bewerberin oder Bewerber
273 		 * @param person     Bewerberin oder Bewerber
274 		 * @return {@code true} if {@code nomination} is the correct nomination for
275 		 *         {@code person}, else {@code false}
276 		 */
277 		private boolean matches(final LocalNomination nomination, final String person) {
278 			final String familyName = nomination.getPerson().getFamilyName();
279 			final String givenName = nomination.getPerson().getGivenName();
280 			return matches(givenName + familyName, person) || matches(familyName + givenName, person);
281 		}

288 		 * @return {@code true} if {@code person} matches {@code value}, else

291 		private boolean matches(final String value, final String person) {
292 			final Pattern pattern = patternCache.computeIfAbsent(person, key -> Pattern.compile(getSimplifiedString(key).chars().mapToObj(v -> Character.toString((char) v)).collect(joining(".*", "^.*", ".*$"))));

359 		/**
360 		 * Formats and writes the {@link #result} of {@link #pollingStation} to
361 		 * {@link #writer}.
362 		 *
363 		 * @throws IOException on IO error
364 		 */
365 		@SuppressWarnings("checkstyle:MultipleStringLiterals")
366 		public void write() throws IOException {
367 			write("%s %s\n", LINE_COMMENT, pollingStation.getName());
368 			final OptionalInt numberOfEligibleVoters = result.getElection().getNumberOfEligibleVoters();
369 			if (numberOfEligibleVoters.isPresent()) {
370 				write("%s Anzahl Wahlberechtigte: %d\n", LINE_COMMENT, numberOfEligibleVoters.getAsInt());
371 			}
372 			final OptionalInt numberOfAllBallots = result.getNumberOfAllBallots(pollingStation);
373 			if (numberOfAllBallots.isPresent()) {
374 				write("%s Anzahl Stimmzettel: %d\n", LINE_COMMAND, numberOfAllBallots.getAsInt());
375 			}
376 			write("\n");
377 			final int numberOfInvaliBallots = result.getNumberOfInvalidBallots();
378 			if (numberOfInvaliBallots > 0) {
379 				write("%s Ungültige Stimmzettel\n", LINE_COMMENT);
380 				write("%d %s\n\n", numberOfInvaliBallots, BALLOT_INVALID);
381 			}
382 			final Collection<LocalPartyResult> partyResults = result.getPartyResults().values();
383 			if (!partyResults.isEmpty()) {
384 				write("%s Blockstimmen\n", LINE_COMMENT);
385 				for (final LocalPartyResult partyResult : partyResults) {
386 					write("%d %s\n", partyResult.getNumberOfBlockVotings(), formatDirectNominations(partyResult.getParty()));
387 				}
388 				write("\n");
389 			}
390 			write("%s Stimmzettel\n", LINE_COMMENT);
391 			for (final LocalBallot ballot : result.getBallots()) {
392 				if (!ballot.isBlockVoting()) {
393 					write("%s\n", formatNominations(ballot.getNominations()));
394 				}
395 			}
396 		}
